\begin{frame}{Answer Set Programming (ASP)}
  \begin{itemize}
  \item<1-> \structure{What is ASP?}  \only<3>{\qquad\hfill\alert{\textbf{ASP}\ =\ \textbf{DB}+\textbf{LP}+\textbf{KR}+\textbf{SAT}\,\textbf{!}}\hfill}

    ASP is an approach for declarative problem solving
  \item <only@2-3> \structure{Where is ASP from?}
      \begin{itemize}
      \item <only@2->  Databases
      \item <only@2->  Logic programming
      \item <only@2->  Knowledge representation and reasoning
      \item <only@2->  Satisfiability solving
      \end{itemize}%
  \item<only@4-> \structure{What is ASP good for?}

    Solving knowledge-intense combinatorial (optimization) problems
  \item<only@5-> \structure{What problems are this?}\only<7-10>{ --- \alert{\bf And industrial ones\,?}}

    \only<5-7,11->{Problems consisting of (many) decisions and constraints}
    \only<6-7>{\smallskip\par\structure{Examples} \
    Sudoku,
    Configuration,
    Diagnosis,
    Music composition,
    Planning,
    System design,
    Time tabling,
    etc.}%
    \only<8-10>{%
      \begin{itemize}
      \item Debian, Ubuntu: Linux package configuration
      \item Exeura: Call routing
      \item \alert<9->{\textbf<9->{Fcc: Radio frequency auction}}
      \only<10>{\begin{picture}(0,0)
        \put(0,5){\vector(1,1){10}}
        \put(82,58){\makebox(0,0){\frame{\includegraphics[width=5cm]{motivation/pictures/fcc}}}}
      \end{picture}}
      \item Gioia Tauro: Workforce management
      \item Nasa: Decision support for Space Shuttle
      \item Siemens: Partner units configuration
      \item Variantum: Product configuration
      \item US Navy: risk assessment
      \end{itemize}
    }
  \item<only@11-> \structure{What are ASP's distinguishing features?}
    \begin{itemize}\normalsize
    \item High level, versatile modeling language
    \item High performance solvers
    \item Qualitative and quantitative optimization
    \end{itemize}
  \item<only@12-> \structure{Any industrial impact?}
    \begin{itemize}
    \item ASP Tech companies: DLV Systems and \alert<12>{\textbf<12>{Potassco Solutions}}
    \item Increasing interest in (large) companies
    \end{itemize}
